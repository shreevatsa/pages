\bigskip On the first day of Christmas my true love gave to me
\par a partridge in a pear tree.

\newtoks\currentGifts
\currentGifts={\par and a partridge in a pear tree.}

\def\christmas #1: #2, #3! {
  \bigskip On the #1 day of Christmas my true love gave to me
  % To mutate \currentGifts we need to expand it so we need \edef (!)
  \edef\modifyGifts{\noexpand\currentGifts={\par #2 \the\currentGifts}}
  \modifyGifts
  \the\currentGifts
  \if #3Z
  \else
    \christmas #3!
  \fi
}
