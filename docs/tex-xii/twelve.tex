\def\gifts #1#2#3#4: #5, #6 {
  % Typeset current gift from #2 and earlier gifts from #3
  \par #1#2#3#5{ing}
  \if #6Z
  \else
    \gifts #6
  \fi
}

\def\christmas #1: #2, #3! {
  % Typeset earlier verses: the ones in #3
  \if #3Z
  \else
    \christmas #3!
  \fi
  % Typeset current verse.
  \bigskip On the #1 day of Christmas my true love gave to me
  \gifts #1: #2, #3
}
