\def\gifts #1: #2, #3! { % Called like: \gifts third: three french hens, second: two turtle doves, Z!
\par #2
\if
    #3Z
\else
    \gifts #3!
\fi
}

\def\christmas #1: #2, #3! { % Called with #1 = fourth, #2 = four calling birds, #3 = <third: three french hens, second: two turtle doves, Z>
\if #3Z
    \bigskip On the #1 day of Christmas my true love gave to me
    \par #2
    \par and a partridge in a pear tree.
\else
    \christmas #3!
    \bigskip On the #1 day of Christmas my true love gave to me
    \par #2
    \gifts #3!
    \par and a partridge in a pear tree.
\fi
}
